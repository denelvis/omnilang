Complete RPC Service
==================

module hello

service GreetingService {
  rpc SayHello {
    inputs:
      name: String
    outputs:
      greeting: Greeting
    preconditions:
      - name != ""
    postconditions:
      - greeting.message != ""
    tests:
      - scenario: "Greet a user"
        given: name == "World"
        expect: greeting.message == "Hello, World!"
  }
}
